ErrorCode {

  0x00;     // 成功

  0x11;     // 指令发送出错
  
  0x12;     // 写入失败
  
  0x13;     // 电量低
  
  0x14;     // 不支持的操作
  
  0x15;     // 不合法
  
  0x16;     // 忙碌中
  
  0x17;     // 未授权
  
  0x18;     // 未知
  
  0x19;     // 设备未绑定
  
  0x1A;     // 正在录音不能对文件进行操作
  
  0x1B;     // 正在播放不能对文件进行操作
 
  0x1c;     // 绑定失败，用户已被绑定
  
  0x1d;     // 用户拒绝绑定

  0x1e;     // 操作失败，操作过于频繁
}
